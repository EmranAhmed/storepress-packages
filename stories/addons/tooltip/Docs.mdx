## Install

```sh
npm i @storepress/tooltip --save
```

## Usage

```js
/**
 * External dependencies
 */
import StorePressTooltip from '@storepress/tooltip';

import '@storepress/tooltip/src/style.scss';

document.addEventListener('DOMContentLoaded', () => {
  // Init tooltip
  StorePressTooltip.init()

  // Destroy tooltip
  StorePressTooltip.destroy()
});
```

## Customized Tooltip

- Add `storepress-tooltip-type-image` class for image tooltip.

```html
<span data-storepress-tooltip="Tooltip text"></span>

- For Image tooltip
<div class="storepress-tooltip-type-image" style="--tooltip-image: url('https://picsum.photos/150')" data-storepress-tooltip="Monotone ctallysyndicatemarketpositioning.">Text</div>
```

```scss
@charset "UTF-8";

@use "~@storepress/tooltip/src/mixins" as tooltip;

:where([data-storepress-tooltip]) {
	@include tooltip.init("data-storepress-tooltip");
}

// Changing colors
// See: "~@storepress/tooltip/src/mixins" variables() mixins.
[data-storepress-tooltip] {
  --tooltip-text-color: #ffffff;
  --tooltip-background-color: #e31616;
}
```
