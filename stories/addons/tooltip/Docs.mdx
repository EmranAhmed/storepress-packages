## Install

```shell
npm i @storepress/tooltip @storepress/utils --save
```

## Usage

```js
/**
 * External dependencies
 */
import StorePressTooltip from '@storepress/tooltip';
import { triggerEvent } from '@storepress/utils';
import '@storepress/tooltip/src/style.scss';

document.addEventListener('DOMContentLoaded', () => {
  // Init tooltip
  StorePressTooltip.init()

  // Destroy tooltip
  StorePressTooltip.destroy()
});
```

## Customized Tooltip

```html
<span class="tooltip" data-storepress-tooltip="Tooltip text"></span>

- For Image tooltip
<div class="storepress-tooltip-type-image" style="--tooltip-image: url('https://placehold.co/300x200/ff5c00/fff')" data-storepress-tooltip="Monotone ctallysyndicatemarketpositioning.">Text</div>
```

```scss

// style.scss

@charset "UTF-8";

@use "~@storepress/tooltip/src/mixins" as tooltip;

:where([data-storepress-tooltip]) {
	@include tooltip.init("data-storepress-tooltip");
}

// Changing colors
// See: "~@storepress/tooltip/src/mixins" variables() mixins.
[data-storepress-tooltip] {
  --tooltip-text-color: #ffffff;
  --tooltip-background-color: #e31616;
}
```
