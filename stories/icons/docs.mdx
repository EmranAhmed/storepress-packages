## Install

```shell
npm i @storepress/icons --save
```

## Usage

```jsx
import { Icon } from '@wordpress/icons';
import { storepress } from '@storepress/icons';

const IconExample = () => <Icon icon={ storepress } />;

function IconExample2() {
	return <Icon icon={storepress} size="48" />
}

export default IconExample2;
```
